235 unsigned int prime_ge ( unsigned int n )
236 {
237  unsigned int lb = 0;
238  unsigned int ub = PRIME_MAX-1;
239  unsigned int min_p = prime(lb);
240  unsigned int max_p = prime(ub);
241  if (n<min_p || n>max_p) {
242  pagmo_throw(value_error,"n is out of range");
243  }
244  if (n == min_p) return min_p;
245  if (n == max_p) return max_p;
246  while(!(ub-lb==1)) {
247  unsigned int tmp = (ub+lb)/2;
248  unsigned int new_p = prime(tmp);
249  if (new_p >= n) {
250  ub = tmp;
251  max_p = new_p;
252  } else {
253  lb = tmp;
254  min_p = new_p;
255  }
256  }
257  return max_p;
258 }

264 int *faure::binomial_table ( int qs, int m, int n )
265 
266 //****************************************************************************80
267 //
268 // Purpose:
269 //
270 // BINOMIAL_TABLE computes a table of bionomial coefficients MOD QS.
271 //
272 // Discussion:
273 //
274 // Thanks to Michael Baudin for pointing out an error in a previous
275 // version of this function, 07 December 2009.
276 //
277 // Licensing:
278 //
279 // This code is distributed under the GNU LGPL license.
280 //
281 // Modified:
282 //
283 // 07 December 2009
284 //
285 // Author:
286 //
287 // John Burkardt
288 //
289 // Parameters:
290 //
291 // Input, int QS, the base for the MOD operation.
292 //
293 // Input, int M, N, the limits of the binomial table.
294 //
295 // Output, int BINOMIAL_TABLE[(M+1)*(N+1)], the table of binomial
296 // coefficients modulo QS.
297 //
298 {
299  int *coef;
300  int i;
301  int j;
302 
303  coef = new int[(m+1)*(n+1)];
304 
305  for ( j = 0; j <= n; j++ )
306  {
307  for ( i = 0; i <= m; i++ )
308  {
309  coef[i+j*(m+1)] = 0;
310  }
311  }
312 
313  coef[0] = 1;
314 
315  j = 0;
316  for ( i = 1; i <= m; i++ )
317  {
318  coef[i+j*(m+1)] = 1;
319  }
320 
321  for ( i = 1; i <= i4_min ( m, n ); i++ )
322  {
323  j = i;
324  coef[i+j*(m+1)] = 1;
325  }
326 
327  for( j = 1; j <= n; j++ )
328  {
329  for ( i = j + 1; i <= m; i++ )
330  {
331  coef[i+j*(m+1)] = ( coef[i-1+j*(m+1)] + coef[i-1+(j-1)*(m+1)] ) % qs;
332  }
333  }
334 
335  return coef;
336 }

339 void faure::faure_orig ( unsigned int dim_num, unsigned int *seed, double quasi[] )
340 
341 //****************************************************************************80
342 //
343 // Purpose:
344 //
345 // FAURE generates a new quasirandom Faure vector with each call.
346 //
347 // Discussion:
348 //
349 // This routine implements the Faure method for computing
350 // quasirandom numbers. It is a merging and adaptation of
351 // Bennett Fox's routines INFAUR and GOFAUR from ACM TOMS Algorithm 647.
352 //
353 // Michael Baudin suggested a change so that whenever HISUM is altered,
354 // the binomial table is recomputed, which allows a user to go back or
355 // forth in the sequence. 08 December 2009.
356 //
357 // Licensing:
358 //
359 // This code is distributed under the GNU LGPL license.
360 //
361 // Modified:
362 //
363 // 08 December 2009
364 //
365 // Author:
366 //
367 // John Burkardt
368 //
369 // Reference:
370 //
371 // Henri Faure,
372 // Discrepance de suites associees a un systeme de numeration
373 // (en dimension s),
374 // Acta Arithmetica,
375 // Volume 41, 1982, pages 337-351.
376 //
377 // Bennett Fox,
378 // Algorithm 647:
379 // Implementation and Relative Efficiency of Quasirandom
380 // Sequence Generators,
381 // ACM Transactions on Mathematical Software,
382 // Volume 12, Number 4, December 1986, pages 362-376.
383 //
384 // Parameters:
385 //
386 // Input, int DIM_NUM, the spatial dimension, which should be
387 // at least 2.
388 //
389 // Input/output, int *SEED, the seed, which can be used to index
390 // the values. On first call, set the input value of SEED to be 0
391 // or negative. The routine will automatically initialize data,
392 // and set SEED to a new value. Thereafter, to compute successive
393 // entries of the sequence, simply call again without changing
394 // SEED. On the first call, if SEED is negative, it will be set
395 // to a positive value that "skips over" an early part of the sequence
396 // (This is recommended for better results).
397 //
398 // Output, double QUASI[DIM_NUM], the next quasirandom vector.
399 //
400 {
401  int hisum;
402  int ktemp;
403  int ltemp;
404  int mtemp;
405  double r;
406  int ztemp;
407 //
408 // Initialization required or requested?
409 //
410  if ( m_qs <= 0 || *seed <= 0 )
411  {
412  m_qs = prime_ge ( dim_num );
413 
414  if ( m_qs < 1 )
415  {
416  cout << "\n";
417  cout << "FAURE - Fatal error!\n";
418  cout << " PRIME_GE failed.\n";
419  exit ( 1 );
420  }
421  m_hisum_save = -1;
422  }
423 
424  if ( *seed == 0 )
425  {
426  hisum = 0;
427  }
428  else
429  {
430  hisum = i4_log_i4 ( *seed, m_qs );
431  }
432 //
433 // Is it necessary to recompute the coefficient table?
434 //
435  if ( m_hisum_save != hisum )
436  {
437  if ( m_coef != NULL )
438  {
439  delete [] m_coef;
440  }
441 
442  if ( m_ytemp != NULL )
443  {
444  delete [] m_ytemp;
445  }
446 
447  m_hisum_save = hisum;
448 
449  m_coef = binomial_table ( m_qs, hisum, hisum );
450 
451  m_ytemp = new int[hisum+1];
452  }
453 //
454 // Find QUASI(1) using the method of Faure.
455 //
456 // SEED has a representation in base QS of the form:
457 //
458 // Sum ( 0 <= J <= HISUM ) YTEMP(J) * QS**J
459 //
460 // We now compute the YTEMP(J)'s.
461 //
462  ktemp = i4_power ( m_qs, hisum + 1 );
463  ltemp = *seed;
464 
465  for ( int i = hisum; 0 <= i; i-- )
466  {
467  ktemp = ktemp / m_qs;
468  mtemp = ltemp % ktemp;
469  m_ytemp[i] = ( ltemp - mtemp ) / ktemp;
470  ltemp = mtemp;
471  }
472 //
473 // QUASI(K) has the form
474 //
475 // Sum ( 0 <= J <= HISUM ) YTEMP(J) / QS**(J+1)
476 //
477 // Compute QUASI(1) using nested multiplication.
478 //
479  r = ( ( double ) m_ytemp[hisum] );
480  for ( int i = hisum-1; 0 <= i; i-- )
481  {
482  r = ( ( double ) m_ytemp[i] ) + r / ( ( double ) m_qs );
483  }
484 
485  quasi[0] = r / ( ( double ) m_qs );
486 //
487 // Find components QUASI(2:DIM_NUM) using the Faure method.
488 //
489  for ( unsigned int k = 1; k < dim_num; k++ )
490  {
491  quasi[k] = 0.0;
492  r = 1.0 / ( ( double ) m_qs );
493 
494  for ( int j = 0; j <= hisum; j++ )
495  {
496  ztemp = 0;
497  for ( int i = j; i <= hisum; i++ )
498  {
499  ztemp = ztemp + m_ytemp[i] * m_coef[i+j*(hisum+1)];
500  }
501 //
502 // New YTEMP(J) is:
503 //
504 // Sum ( J <= I <= HISUM ) ( old ytemp(i) * binom(i,j) ) mod QS.
505 //
506  m_ytemp[j] = ztemp % m_qs;
507  quasi[k] = quasi[k] + ( ( double ) m_ytemp[j] ) * r;
508  r = r / ( ( double ) m_qs );
509  }
510  }
511 //
512 // Update SEED.
513 //
514  *seed = *seed + 1;
515 
516  return;
517 }

522 //****************************************************************************80
523 //
524 // Purpose:
525 //
526 // I4_LOG_I4 returns the logarithm of an I4 to an I4 base.
527 //
528 // Discussion:
529 //
530 // Only the integer part of the logarithm is returned.
531 //
532 // If
533 //
534 // K4 = I4_LOG_J4 ( I4, J4 ),
535 //
536 // then we ordinarily have
537 //
538 // J4^(K4-1) < I4 <= J4^K4.
539 //
540 // The base J4 should be positive, and at least 2. If J4 is negative,
541 // a computation is made using the absolute value of J4. If J4 is
542 // -1, 0, or 1, the logarithm is returned as 0.
543 //
544 // The number I4 should be positive and at least 2. If I4 is negative,
545 // a computation is made using the absolute value of I4. If I4 is
546 // -1, 0, or 1, then the logarithm is returned as 0.
547 //
548 // An I4 is an integer ( kind = 4 ) value.
549 //
550 // Example:
551 //
552 // I4 J4 K4
553 //
554 // 0 3 0
555 // 1 3 0
556 // 2 3 0
557 // 3 3 1
558 // 4 3 1
559 // 8 3 1
560 // 9 3 2
561 // 10 3 2
562 //
563 // Licensing:
564 //
565 // This code is distributed under the GNU LGPL license.
566 //
567 // Modified:
568 //
569 // 09 June 2007
570 //
571 // Author:
572 //
573 // John Burkardt
574 //
575 // Parameters:
576 //
577 // Input, int I4, the number whose logarithm is desired.
578 //
579 // Input, int J4, the base of the logarithms.
580 //
581 // Output, int I4_LOG_I4, the integer part of the logarithm
582 // base abs(J4) of abs(I4).
583 //
584 {
585  int i4_abs;
586  int j4_abs;
587  int value;
588 
589  value = 0;
590 
591  i4_abs = abs ( i4 );
592 
593  if ( 2 <= i4_abs )
594  {
595  j4_abs = abs ( j4 );
596 
597  if ( 2 <= j4_abs )
598  {
599  while ( j4_abs <= i4_abs )
600  {
601  i4_abs = i4_abs / j4_abs;
602  value = value + 1;
603  }
604  }
605  }
606  return value;
607 }
