77 tsp::tsp(const std::vector<std::vector<double> > &weights):
78  base_aco(boost::numeric_cast<int>(weights[0].size()),1,0), m_weights(weights) {
79 
80  //Check weights matrix
81  for(problem::base_aco::size_type i = 0; i < m_weights.size(); ++i) {
82  if (m_weights[i].size() != m_weights.size()) {
83  pagmo_throw(value_error,"Weights matrix must be a square matrix!");
84  }
85  if(m_weights[i][i] != 0) {
86  pagmo_throw(value_error,"Weights matrix must have 0's on the diagonal!");
87  }
88  for (problem::base_aco::size_type j = 0; j < m_weights[i].size(); ++j) {
89  if(m_weights[i][j] != m_weights[j][i]) {
90  pagmo_throw(value_error,"Weights matrix must be a simmetric matrix!");
91  }
92  }
93  }
94 
95  set_lb(0);
96  set_ub(weights[0].size()-1); //number of nodes in the graph -1 (we count from 0)
97  set_heuristic_information_matrix();
98 }

120 bool tsp::check_partial_feasibility(const decision_vector &x) const{
121  if (x.size() > get_i_dimension()) {
122  pagmo_throw(value_error,"Invalid chromosome length for partial feasibility check.");
123  }
124 
125  m_tmpDecisionVector = x;
126  std::sort(m_tmpDecisionVector.begin(), m_tmpDecisionVector.end());
127 
128  return std::unique(m_tmpDecisionVector.begin(), m_tmpDecisionVector.end()) == m_tmpDecisionVector.end();

140 void tsp::objfun_impl(fitness_vector &f, const decision_vector &x) const
141 {
142  pagmo_assert(f.size() == 1);
143  pagmo_assert(x.size() == get_dimension() && x.size() == m_weights[0].size());
144  f[0] = 0;
145  for (size_type i = 1; i < get_dimension(); ++i) {
146  f[0] += m_weights[boost::numeric_cast<int>(x[i-1])][boost::numeric_cast<int>(x[i])];
147  }
148  f[0] += m_weights[boost::numeric_cast<int>(x[get_dimension()-1])][boost::numeric_cast<int>(x[0])];
149 }
